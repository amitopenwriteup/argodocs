#create yml file : apphelmgit.yml

apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: my-nginx-amit
  namespace: argocd
spec:
  destination:
    server: https://kubernetes.default.svc
    namespace: stage-my-amit-helm
  sources:
  - repoURL: 'https://charts.bitnami.com/bitnami'
    chart: nginx
    targetRevision: 18.1.2
    helm:
      valueFiles:
      - $values/nginx-helm/values.yaml
  - repoURL: 'https://github.com/amitopenwriteup/nginx-proj.git'
    targetRevision: main
    ref: values
  project: default
  syncPolicy:
    automated:
      prune: true
      selfHeal: true
    syncOptions:
      - CreateNamespace=true

#create yml file

kubectl create -f apphelmgit.yml
